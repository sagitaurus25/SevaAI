        // Configuration

        const API_ENDPOINT = 'https://zfx225n217.execute-api.us-east-1.amazonaws.com/prod';

        function setWelcomeTime() {

        function addMessage(content, type = 'bot', sender = 'Seva') {

            const messagesContainer = document.getElementById('messages');

            const messageDiv = document.createElement('div');

            messageDiv.className = `message ${type}`;

            

            const avatar = document.createElement('div');

            avatar.className = 'message-avatar';

            avatar.textContent = type === 'user' ? '👤' : (type === 'error' ? '❌' : '🤖');

            

            const contentDiv = document.createElement('div');

            contentDiv.className = 'message-content';

            contentDiv.innerHTML = content;

            

            const timeDiv = document.createElement('div');

            timeDiv.className = 'message-time';

            timeDiv.textContent = new Date().toLocaleTimeString();

            contentDiv.appendChild(timeDiv);

            

            messageDiv.appendChild(avatar);

            messageDiv.appendChild(contentDiv);

            messagesContainer.appendChild(messageDiv);

            

            // Scroll to bottom

            messagesContainer.scrollTop = messagesContainer.scrollHeight;

        }

        function setSendButtonState(loading = false) {

            } else {

                button.disabled = false;

                input.disabled = false;

        }


        async function sendMessage() {

            const input = document.getElementById('messageInput');

            const message = input.value.trim();

            

            if (!message) {

                input.focus();

                return;

            }

            

            // Add user message

            addMessage(message, 'user', 'You');

            input.value = '';

            

            // Show loading state

            setSendButtonState(true);

            showTyping();

            setConnectionStatus('connecting', '🟡 Sending...');

            

            try {

                const response = await fetch(API_ENDPOINT, {

                    method: 'POST',

                    headers: {

                        'Content-Type': 'application/json',

                        'Accept': 'application/json'

                    },

                    body: JSON.stringify({

                        message: message,

                        session_id: sessionId,

                        ai_model: document.getElementById('aiModel').value

                    })

                });

                

                if (!response.ok) {

                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);

                }

                

                const data = await response.json();

                console.log('API Response:', data);

                

                hideTyping();

                

                if (data.response) {

                    // Format the response with proper line breaks and styling

                    const formattedResponse = formatResponse(data.response);

                    addMessage(formattedResponse, 'bot', 'Seva');

                    setConnectionStatus('connected', '🟢 Connected');

                } else if (data.error) {

                    addMessage(`<strong>Error:</strong> ${data.error}`, 'error', 'System');

                    setConnectionStatus('error', '🔴 Error');

                } else {

                    addMessage(`<strong>Unexpected response:</strong><br><pre>${JSON.stringify(data, null, 2)}</pre>`, 'error', 'System');

                    setConnectionStatus('error', '🔴 Unexpected');

                }

                

            } catch (error) {

                hideTyping();

                console.error('Connection error:', error);

                addMessage(

                    `<strong>Connection Error:</strong><br>

                    ${error.message}<br><br>

                    <small>Please check your internet connection and try again. If the problem persists, the API might be temporarily unavailable.</small>`, 

                    'error', 

                    'System'

                );

                setConnectionStatus('error', '🔴 Connection Failed');

            } finally {

                setSendButtonState(false);

            }

        }


        function formatResponse(response) {

            // Convert newlines to HTML breaks

            let formatted = response.replace(/\n/g, '<br>');

            

            // Make text between ** bold

            formatted = formatted.replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>');

            

            // Make text between * italic

            formatted = formatted.replace(/\*(.*?)\*/g, '<em>$1</em>');

            

            // Convert URLs to clickable links

            formatted = formatted.replace(/(https?:\/\/[^\s<]+)/g, '<a href="$1" target="_blank">$1</a>');

            

            return formatted;

        }


        function clearChat() {

            if (confirm('Are you sure you want to clear the chat history?')) {

                const messagesContainer = document.getElementById('messages');

                messagesContainer.innerHTML = `

                    <div class="message bot">

                        <div class="message-avatar">🤖</div>

                        <div class="message-content">

                            Chat cleared! What Seva can I do for you?

                            <div class="message-time">${new Date().toLocaleTimeString()}</div>

                        </div>

                    </div>

                `;

                document.getElementById('messageInput').focus();

            }

        }

                e.preventDefault();

                sendMessage();

            }

        });


        // Handle AI model change

        document.getElementById('aiModel').addEventListener('change', function() {

            const selected = this.value;

            const modelNames = {

                'auto': 'Auto (Nova → Claude)',

                'nova': 'Nova Micro Only',

                'claude': 'Claude Sonnet Only'

            };

            

            addMessage(

                `🔧 <strong>AI Model changed to:</strong> ${modelNames[selected]}`, 

                'bot', 

                'System'

            );

        });


        // Add some helpful keyboard shortcuts

        document.addEventListener('keydown', function(e) {

            // Ctrl/Cmd + K to clear chat

            if ((e.ctrlKey || e.metaKey) && e.key === 'k') {

                e.preventDefault();

                clearChat();

            }

            

            // Escape to focus input

            if (e.key === 'Escape') {

                document.getElementById('messageInput').focus();

            }

        });

        setTimeout(async () => {

            try {

                const response = await fetch(API_ENDPOINT, {
